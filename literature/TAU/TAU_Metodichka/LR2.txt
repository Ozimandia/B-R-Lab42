k=3    //параметры варианта

cc=0.7  //коэфф. демпфирования
c=%pi
a1=1/(T*T)
a2=2*cc/T
a3=k*c/(T*T)
function H=f(y)
  if y<-a then
     H=-c
  elseif y<=a
     H=0
  else 
     H=c
  end
endfunction
function F=gb(t,x)
  F=[x(2);x(3);-a1*x(2)-a2*x(3)-a3*f(x(1))]
endfunction
